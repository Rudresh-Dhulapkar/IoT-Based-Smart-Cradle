#include <Servo.h>

Servo myservo;  // create servo object to control a servo
// twelve servo objects can be created on most boards


void setup() {
  pinMode(A0,INPUT);
  Serial.begin(9600);
  
  myservo.attach(5);  
  myservo.write(0);

  delay(2000);
}

void loop() {
  int moisture=analogRead(A0);
  Serial.println(moisture);
  delay(100);
  
  myservo.write(90);
    delay(1000);

  myservo.write(0);
    delay(1000);
}
